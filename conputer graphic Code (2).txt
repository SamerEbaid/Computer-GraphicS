float addy = 0;
float turn = 0;


using std::pair;
pair<float, float> rotate(float x,float y,float turn) {

    float x2 = (x - 3) * cos(turn) - (y - 2) * sin(turn)+3;
    float y2 = (x - 3) * sin(turn) + (y - 2) * cos(turn) +2;

    return { x2,y2 };
}


void display();
void reshape(int w,int h);
float i, angle ;
void init(){
    glClearColor(0.0,0.0,0.0,1.0);
}
void reshape(int w,int h)
{
    glViewport(0,0,(GLsizei)w,(GLsizei)h);
    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    gluOrtho2D(-10,10,-10,10);
    glMatrixMode(GL_MODELVIEW);
}
void timer(int)
{
    glutPostRedisplay();
    glutTimerFunc(1000 / 60, timer, 0);
    step += move; // car
    step2+=move2; // cloud
    if (step>8 ||  step<-2) {   // to make it move from range 8 to -2
        move = -move;
    }
    if (step2>14 ||  step2<0) { // to make it move from range 14 to 0
        move2 = -move2;
    }
    turn += edit_turn;


}





int main(int argc,char**argv)
{
    glutInit(&argc,argv);

    glutCreateWindow("Mariam Mohamed 200100755,Samer Ahmed 20101589");

    glutDisplayFunc(display);
    glutReshapeFunc(reshape);
    glutTimerFunc(0, timer, 0);
    init();

    glutMainLoop();
}
